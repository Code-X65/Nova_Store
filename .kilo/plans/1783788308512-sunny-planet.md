## Phase 3 — Backend Controller Layer

### 3.1 `Backend/src/controllers/admin/rider.controller.js` — extend

**New endpoints in the controller class:**

```js
async approveRider(req, res, next) {
  const { id } = req.params;
  // Permission gate: rider:approve
  const rider = await RiderModel.findById(id);
  if (!rider || rider.status !== 'pending_approval') {
    return res.status(400).json({ success: false, message: 'Rider is not pending approval' });
  }
  const updated = await RiderModel.approve(id, req.admin.id);
  // Fire notification to the rider's creator
  AuditService.log(req, 'rider.approved', 'rider', id);
  res.json({ success: true, data: updated, message: 'Rider approved and now live' });
}

async rejectRider(req, res, next) {
  const { id } = req.params;
  const { rejection_reason } = req.body;
  // Permission gate: rider:approve
  const rider = await RiderModel.findById(id);
  if (!rider || rider.status !== 'pending_approval') {
    return res.status(400).json({ success: false, message: 'Rider is not pending approval' });
  }
  const updated = await RiderModel.update(id, { status: 'pending_approval', rejection_reason });
  AuditService.log(req, 'rider.rejected', 'rider', id);
  res.json({ success: true, data: updated, message: 'Rider rejected' });
}

async suspendRider(req, res, next) {
  const { id } = req.params;
  // Permission gate: rider:suspend
  const rider = await RiderModel.findById(id);
  if (!rider) return res.status(404).json(...);
  const updated = await RiderModel.suspend(id);
  AuditService.log(req, 'rider.suspended', 'rider', id);
  res.json({ success: true, data: updated });
}

async reactivateRider(req, res, next) {
  const { id } = req.params;
  // Permission gate: rider:suspend
  const updated = await RiderModel.update(id, { status: 'live' });
  AuditService.log(req, 'rider.reactivated', 'rider', id);
  res.json({ success: true, data: updated });
}

// Guarantor CRUD
async listGuarantors(req, res, next) {
  const { riderId } = req.params;
  const guarantors = await RiderGuarantorModel.findByRiderId(riderId);
  res.json({ success: true, data: guarantors });
}

async createGuarantor(req, res, next) {
  const { riderId } = req.params;
  const isFull = await RiderGuarantorModel.isFull(riderId);
  if (isFull) return res.status(400).json({ success: false, message: 'Maximum 2 guarantors allowed' });
  const guarantor = await RiderGuarantorModel.create({ ...req.body, rider_id: riderId });
  res.status(201).json({ success: true, data: guarantor });
}

async updateGuarantor(req, res, next) {
  const { guarantorId } = req.params;
  const guarantor = await RiderGuarantorModel.update(guarantorId, req.body);
  res.json({ success: true, data: guarantor });
}

async deleteGuarantor(req, res, next) {
  const { guarantorId } = req.params;
  await RiderGuarantorModel.delete(guarantorId);
  res.json({ success: true, message: 'Guarantor removed' });
}

async getPendingRiders(req, res, next) {
  const riders = await RiderModel.findPending();
  res.json({ success: true, data: riders });
}
```

### 3.2 Update `createRider` and `updateRider` methods

- Enforce: `photo_frontal`, `photo_left_profile`, `photo_right_profile` required on create
- Guarantor limit enforced by model, not controller
- Secondary phone and doc URLs optional
- Structured address fields stored both in new columns and `address_jsonb`
- Status always `'pending_approval'` on create (MANAGER creates → deferred to STORE_OWNER)
- Only STORE_OWNER can set `status: 'live'` directly in update (bypass approval) — use `approveRider` endpoint normally
- `is_active` flag retained; `status='live' AND is_active=true` means rider visible for dispatch

## Phase 13 — Risk & Dependency Analysis

| Risk | Mitigation |
|------|-----------|
| Google Maps API key not configured in production | `GMapAutocomplete` degrades to static dropdowns + manual input; no hard dependency |
| Existing `address_jsonb` data migration | Migration keeps both old JSONB and new columns; on next rider save, both are populated |
